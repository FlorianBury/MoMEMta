241  virtual Status work() override {
242  static std::vector<LorentzVector> empty_vector;
243 
244  *m_integrand = 0;
245  const std::vector<LorentzVector>& partons = *m_partons;
246 
247  LorentzVectorRefCollection particles;
248  for (const auto& p: m_particles)
249  particles.push_back(std::ref(*p));
250 
251  for (size_t i = 0; i < m_particles_ids.size(); i++) {
252  finalState[i] = std::make_pair(m_particles_ids[i].pdg_id, toVector(particles[i].get()));
253  }
254 
255  // Sort the array taking into account the indexing in the configuration
256  apply_permutations(finalState, permutations);
257 
258  std::pair<std::vector<double>, std::vector<double>> initialState { toVector(partons[0]),
259  toVector(partons[1]) };
260 
261  auto result = m_ME->compute(initialState, finalState);
262 
263  double x1 = std::abs(partons[0].Pz() / (sqrt_s / 2.));
264  double x2 = std::abs(partons[1].Pz() / (sqrt_s / 2.));
265 
266  // Compute flux factor 1/(2*x1*x2*s)
267  double phaseSpaceIn = 1. / (2. * x1 * x2 * SQ(sqrt_s));
268 
269  double integrand = phaseSpaceIn;
270  for (const auto& jacobian: m_jacobians) {
271  integrand *= (*jacobian);
272  }
273 
274  // PDF
275  double final_integrand = 0;
276  for (const auto& me: result) {
277  double pdf1 = use_pdf ? m_pdf->xfxQ2(me.first.first, x1, pdf_scale_squared) / x1 : 1;
278  double pdf2 = use_pdf ? m_pdf->xfxQ2(me.first.second, x2, pdf_scale_squared) / x2 : 1;
279 
280  final_integrand += me.second * pdf1 * pdf2;
281  }
282 
283  final_integrand *= integrand;
284  *m_integrand = final_integrand;
285 
286  return Status::OK;
287  }
