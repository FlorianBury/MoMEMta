38 ConfigurationReader::ConfigurationReader(const std::string& file, const ParameterSet& parameters) {
39 
40  LOG(debug) << "Parsing LUA configuration from " << file;
41  lua_state = lua::init_runtime(this);
42 
43  lua::inject_parameters(lua_state.get(), parameters);
44 
45  // Parse file
46  if (luaL_dofile(lua_state.get(), file.c_str())) {
47  std::string error = lua_tostring(lua_state.get(), -1);
48  LOG(fatal) << "Failed to parse configuration file: " << error;
49  throw lua::invalid_configuration_file(error);
50  }
51 
52  // FIXME: Find a better way of doing that
53 
54  // Read global parameters from global variable named 'parameters'
55  configuration.global_parameters.reset(new lua::LazyTable(lua_state, "parameters"));
56  int type = lua_getglobal(lua_state.get(), "parameters");
57  if (type == LUA_TTABLE) {
58  LOG(debug) << "Parsing global parameters.";
59  ParameterSetParser::parse(*configuration.global_parameters, lua_state.get(), -1);
60  }
61  lua_pop(lua_state.get(), 1);
62 
63  // Read cuba configuration
64  configuration.cuba_configuration.reset(new lua::LazyTable(lua_state, "cuba"));
65  type = lua_getglobal(lua_state.get(), "cuba");
66  if (type == LUA_TTABLE) {
67  LOG(debug) << "Parsing cuba configuration.";
68  ParameterSetParser::parse(*configuration.cuba_configuration, lua_state.get(), -1);
69  }
70  lua_pop(lua_state.get(), 1);
71 
72  for (auto& m: configuration.modules) {
73  LOG(debug) << "Module declared: " << m.type << "::" << m.name;
74 
75  lua_getglobal(lua_state.get(), m.type.c_str());
76  lua_getfield(lua_state.get(), -1, m.name.c_str());
77 
78  m.parameters.reset(new ParameterSet(m.type, m.name));
79  ParameterSetParser::parse(*m.parameters, lua_state.get(), -1);
80 
81  lua_pop(lua_state.get(), 2);
82  }
83 }
