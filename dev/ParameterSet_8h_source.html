274  template<typename T>
275  void set_helper(const std::string& name, const T& value) {
276  static_assert(
277  std::is_same<T, int64_t>::value ||
278  std::is_same<T, double>::value ||
279  std::is_same<T, bool>::value ||
280  std::is_same<T, std::string>::value ||
281  std::is_same<T, InputTag>::value,
282  "Type not supported"
283  );
284 
285  if (frozen) {
286  LOG(fatal) << "You are not allowed to edit a set once frozen.";
287  throw frozen_error("This ParameterSet is frozen");
288  }
289 
290  auto it = m_set.find(name);
291  // If the element does not exist in the set, we create it
292  // Otherwise, we simply update the value
293  if (it == m_set.end()) {
294  create(name, value);
295  } else {
296  setInternal(name, it->second, value);
297  }
298  }
