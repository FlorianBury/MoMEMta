95 Configuration Configuration::freeze() const {
96  Configuration c = *this;
97 
98  c.global_parameters->freeze();
99  c.cuba_configuration->freeze();
100  for (auto& module: c.modules) {
101  module.parameters->freeze();

104  }
105
