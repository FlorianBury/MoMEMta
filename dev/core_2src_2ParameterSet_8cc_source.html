31 void ParameterSet::parse(lua_State* L, int index) {
32 
33  LOG(trace) << "[parse] >> stack size = " << lua_gettop(L);
34  size_t absolute_index = lua::get_index(L, index);
35 
36  lua_pushnil(L);
37  while (lua_next(L, absolute_index) != 0) {
38 
39  std::string key = lua_tostring(L, -2);
40 
41  LOG(trace) << "[parse] >> key = " << key;
42 
43  try {
44  boost::any value;
45  bool lazy = false;
46  std::tie(value, lazy) = parseItem(key, L, -1);
47 
48  m_set.emplace(key, Element(value, lazy));
49  } catch(...) {
50  LOG(fatal) << "Exception while trying to parse parameter " << getModuleType() << "." << getModuleName() << "::" << key;
51  lua_pop(L, 1);
52  std::rethrow_exception(std::current_exception());
53  }
54 
55  lua_pop(L, 1);
56  }
57 
58  LOG(trace) << "[parse] << stack size = " << lua_gettop(L);
59 }

83 void ParameterSet::freeze() {
84  if (frozen)
85  return;
86 
87  frozen = true;
88 
89  for (auto& p: m_set) {
90  auto& element = p.second;
91  try {
92  if (element.lazy) {
93  element.lazy = false;
94  if (element.value.type() == typeid(lua::LazyFunction)) {
95  element.value = boost::any_cast<lua::LazyFunction>(element.value)();
96  } else if (element.value.type() == typeid(lua::LazyTableField)) {
97  element.value = boost::any_cast<lua::LazyTableField>(element.value)();
98  }
99  } else {
100  // Recursion
101  if (element.value.type() == typeid(ParameterSet)) {
102  ParameterSet& s = boost::any_cast<ParameterSet&>(element.value);
103  s.freeze();
104  } else if (element.value.type() == typeid(std::vector<ParameterSet>)) {
105  std::vector<ParameterSet>& v = boost::any_cast<std::vector<ParameterSet>&>(element.value);
106  for (auto& c: v)
107  c.freeze();
108  }
109  }
110  } catch(...) {
111  LOG(fatal) << "Exception while trying to parse parameter " << getModuleType() << "." << getModuleName() << "::" << p.first;
112  std::rethrow_exception(std::current_exception());
113  }
114  }
115 }
