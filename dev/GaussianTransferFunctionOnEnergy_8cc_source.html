93  public:
94  GaussianTransferFunctionOnEnergy(PoolPtr pool, const ParameterSet& parameters): GaussianTransferFunctionOnEnergyBase(pool, parameters) {
95  m_ps_point = get<double>(parameters.get<InputTag>("ps_point"));
96  }

98  virtual Status work() override {
99  // Estimate the width over which to integrate using the width of the TF at E_rec ...
100  const double sigma_E_rec = m_reco_input->E() * m_sigma;
101 
102  double range_min = std::max(m_reco_input->M(), m_reco_input->E() - (m_sigma_range * sigma_E_rec));
103  double range_max = m_reco_input->E() + (m_sigma_range * sigma_E_rec);
104  double range = (range_max - range_min);
105 
106  double gen_E = range_min + range * (*m_ps_point);
107  double gen_pt = std::sqrt(SQ(gen_E) - SQ(m_reco_input->M())) / std::cosh(m_reco_input->Eta());
108 
109  output->SetCoordinates(
110  gen_pt * std::cos(m_reco_input->Phi()),
111  gen_pt * std::sin(m_reco_input->Phi()),
112  gen_pt * std::sinh(m_reco_input->Eta()),
113  gen_E);
114 
115  // ... but compute the width of the TF at E_gen!
116  const double sigma_E_gen = gen_E * m_sigma;
117 
118  // Compute TF*jacobian, where the jacobian includes the transformation of [0,1]->[range_min,range_max] and d|P|/dE
119  *TF_times_jacobian = ROOT::Math::normal_pdf(gen_E, sigma_E_gen, m_reco_input->E()) * range * dP_over_dE(*output);
120 
121  return Status::OK;
122  }
123 
124  private:
125  // Input
126  Value<double> m_ps_point;
127 
128  // Outputs
129  std::shared_ptr<LorentzVector> output = produce<LorentzVector>("output");
130  std::shared_ptr<double> TF_times_jacobian = produce<double>("TF_times_jacobian");
131 
132 };
