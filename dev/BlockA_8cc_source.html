84  virtual Status work() override {
85 
86  solutions->clear();
87 
88  const LorentzVector& p1 = *m_particles[0];
89  const LorentzVector& p2 = *m_particles[1];
90 
91  LorentzVector pb;
92  for (size_t i = 2; i < m_particles.size(); i++) {
93  pb += *m_particles[i];
94  }
95 
96  double pbx = pb.Px();
97  double pby = pb.Py();
98  const double theta1 = p1.Theta();
99  const double phi1 = p1.Phi();
100  const double theta2 = p2.Theta();
101  const double phi2 = p2.Phi();
102  const double m1 = p1.M();
103  const double m2 = p2.M();
104 
105  // pT = p1+p2+pb = 0. Equivalent to the following system:
106  // p1x+p2x = -pbx
107  // p1y+p2y = -pby,
108  // where: p1x=modp1*sin(theta1)*cos(phi1), p1y=modp1*sin(theta1)*sin(phi1),
109  // p2x=modp2*sin(theta2)*cos(phi2), p2y=modp2*sin(theta2)*sin(phi2)
110  // Get modp1, modp2 as solutions of this system
111 
112  std::vector<double> modp1;
113  std::vector<double> modp2;
114 
115  const double sin_theta1 = std::sin(theta1);
116  const double cos_phi1 = std::cos(phi1);
117  const double sin_theta2 = std::sin(theta2);
118  const double cos_phi2 = std::cos(phi2);
119  const double sin_phi1 = std::sin(phi1);
120  const double sin_phi2 = std::sin(phi2);
121  const double cos_theta1 = std::cos(theta1);
122  const double cos_theta2 = std::cos(theta2);
123 
124  const double a10 = sin_theta1 * cos_phi1;
125  const double a01 = sin_theta2 * cos_phi2;
126  const double a00 = pbx;
127  const double b10 = sin_theta1 * sin_phi1;
128  const double b01 = sin_theta2 * sin_phi2;
129  const double b00 = pby;
130 
131  bool foundSolution = solve2Linear(a10, a01, a00, b10, b01, b00, modp1, modp2, false);
132 
133  if (!foundSolution)
134  return Status::NEXT;
135 
136  double mod_p1 = modp1[0];
137  double mod_p2 = modp2[0];
138  if (mod_p1 < 0 || mod_p2 < 0)
139  return Status::NEXT;
140 
141  double E1 = sqrt(SQ(mod_p1) + SQ(m1));
142  double E2 = sqrt(SQ(mod_p2) + SQ(m2));
143 
144  LorentzVector gen_p1(mod_p1 * sin_theta1 * cos_phi1, mod_p1 * sin_theta1 * sin_phi1, mod_p1 * cos_theta1, E1);
145  LorentzVector gen_p2(mod_p2 * sin_theta2 * cos_phi2, mod_p2 * sin_theta2 * sin_phi2, mod_p2 * cos_theta2, E2);
146 
147  // Check if solutions are physical
148  LorentzVector tot = gen_p1 + gen_p2;
149  for (size_t i = 0; i < m_particles.size(); i++) {
150  tot += *m_particles[i];
151  }
152  double q1Pz = std::abs(tot.Pz() + tot.E()) / 2.;
153  double q2Pz = std::abs(tot.Pz() - tot.E()) / 2.;
154  if (q1Pz > sqrt_s / 2 || q2Pz > sqrt_s / 2)
155  return Status::NEXT;
156 
157  double jacobian = (SQ(mod_p1) * SQ(mod_p2)) / (8 * SQ(M_PI * sqrt_s) * E1 * E2);
158  jacobian *= 1. / std::abs(cos_phi1 * sin_phi2 - sin_phi1 * cos_phi2);
159 
160  Solution s = { {gen_p1, gen_p2}, jacobian, true };
161  solutions->push_back(s);
162 
163  return Status::OK;
164  }
