37 
38  // Initialize shared memory pool for modules
39  m_pool.reset(new Pool());
40 
41  // Create phase-space points vector, input for many modules
42  m_pool->current_module("cuba");
43  m_ps_points = m_pool->put<std::vector<double>>({"cuba", "ps_points"});
44  m_ps_weight = m_pool->put<double>({"cuba", "ps_weight"});
45 
46  // Create vector for input particles
47  m_pool->current_module("input");
48  m_particles = m_pool->put<std::vector<LorentzVector>>({"input", "particles"});
49  // Create input for met
50  m_met = m_pool->put<LorentzVector>({"input", "met"});
51 
52  // Construct modules from configuration
53  std::vector<Configuration::Module> light_modules = configuration.getModules();
54  for (const auto& module: light_modules) {
55  m_pool->current_module(module);
56  m_modules.push_back(ModuleFactory::get().create(module.type, m_pool, *module.parameters));
57  m_modules.back()->configure();
58  }
59 
60  m_n_dimensions = configuration.getNDimensions();
61  LOG(info) << "Number of dimensions for integration: " << m_n_dimensions;
62 
63  // Resize pool ps-points vector
64  m_ps_points->resize(m_n_dimensions);
65 
66  // Integrand
67  // First, check if the user defined which integrand to use
68  std::vector<InputTag> integrands = configuration.getIntegrands();
69  if (!integrands.size()) {
70  LOG(fatal) << "No integrand found. Define which module's output you want to use as the integrand using the lua `integrand` function.";
71  throw integrands_output_error("No integrand found");
72  }
73 
74  // Next, retrieve all the input tags for the components of the integrand
75  m_pool->current_module("momemta");
76 
77  for(const auto& component: integrands) {
78  m_integrands.push_back(m_pool->get<double>(component));
79  LOG(debug) << "Configuration declared integrand component using: " << component.toString();
80  }
81  m_n_components = m_integrands.size();
82 
83  m_cuba_configuration = configuration.getCubaConfiguration();
84 
85  const Pool::DescriptionMap& description = m_pool->description();
86  graph::build(description, m_modules, configuration.getPaths(), [&description, this](const std::string& module) {
87  // Clean the pool for each removed module
88  const Description& d = description.at(module);
89  for (const auto& input: d.inputs)
90  this->m_pool->remove_if_invalid(input);
91  for (const auto& output: d.outputs)
92  this->m_pool->remove({module, output});
93  });
94 
95  // Freeze the pool after removing unneeded modules
96  m_pool->freeze();
97 
98  cubacores(0, 0);
99 
100  // Register logging function
101  cubalogging(MoMEMta::cuba_logging);
102 }

180  grid_file.c_str(), // (char*) name of state file => state can be stored and retrieved for further refinement
181  NULL, // (int*) "spinning cores": -1 || NULL <=> integrator takes care of starting & stopping child processes (other value => keep or retrieve child processes, probably not useful here)
182  &neval, // (int*) actual number of evaluations done
183  &nfail, // 0=desired accuracy was reached; -1=dimensions out of range; >0=accuracy was not reached
184  mcResult.get(), // (double*) integration result ([ncomp])
185  error.get(), // (double*) integration error ([ncomp])
186  prob.get() // (double*) Chi-square p-value that error is not reliable (ie should be <0.95) ([ncomp])
187  );
188 
189  if (nfail == 0) {
190  integration_status = IntegrationStatus::SUCCESS;
191  } else if (nfail == -1) {
192  integration_status = IntegrationStatus::DIM_OUT_OF_RANGE;
193  } else if (nfail > 0) {
194  integration_status = IntegrationStatus::ACCURACY_NOT_REACHED;

198  } else {

206  integration_status = IntegrationStatus::SUCCESS;
207  } else {
208  integration_status = IntegrationStatus::ABORTED;
209  }
210  }
211 
212  for (const auto& module: m_modules) {
213  module->endIntegration();
214  }
215 
216  std::vector<std::pair<double, double>> result;
217  for (size_t i = 0; i < m_n_components; i++) {
218  result.push_back( std::make_pair(mcResult[i], error[i]) );

224 int MoMEMta::integrand(const double* psPoints, const double* weights, double* results) {
225 
226  // Store phase-space points into the pool
227  std::memcpy(m_ps_points->data(), psPoints, sizeof(double) * m_n_dimensions);
228 
229  if (weights != nullptr) {
230  // Store phase-space weight into the pool
231  *m_ps_weight = *weights;
232  }
233 
234  for (auto& module: m_modules)
235  module->beginPoint();
236 
237  for (auto& module: m_modules) {
238  auto status = module->work();
239 
240  if (status == Module::Status::NEXT) {
241  // Stop executation for the current integration step
242  // Returns 0 so that cuba knows this phase-space volume is not relevant
243  for (size_t i = 0; i < m_n_components; i++)
244  results[i] = 0;
245  return CUBA_OK;
246  } else if (status == Module::Status::ABORT) {
247  // Abort integration
248  for (size_t i = 0; i < m_n_components; i++)
249  results[i] = 0;
250  return CUBA_ABORT;
251  }
252  }
253 
254  for (auto& module: m_modules)
255  module->endPoint();
256 
257  for (size_t i = 0; i < m_n_components; i++)
258  results[i] = *(m_integrands[i]);
259 
260  return CUBA_OK;
261 }
