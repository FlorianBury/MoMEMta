38 momemta::any Pool::reserve(const InputTag& tag) {
39  auto it = m_storage.find(tag);
40  if (it == m_storage.end()) {
41  // This tag do not exist currently in the pool
42  // Reserve a slot, but mark it as invalid.
43  // Once a module inform the pool it produces such a tag, the slot will
44  // be flagged as valid.
45  PoolContent content { momemta::any(), false };
46  it = m_storage.emplace(tag, content).first;
47  }
48 
49  // Update current module description
50  Description& description = get_description();
51  description.inputs.push_back(tag);
52 
53  return it->second.ptr;
54 }

56 void Pool::alias(const InputTag& from, const InputTag& to) {
57  if (from.isIndexed() || to.isIndexed()) {
58  throw std::invalid_argument("Indexed input tag cannot be passed as argument of the pool. Use the `get` function of the input tag to retrieve its content.");
59  }
60 
61  auto from_it = m_storage.find(from);
62  if (from_it == m_storage.end())
63  throw tag_not_found_error("No such tag in pool: " + from.toString());
64 
65  auto to_it = m_storage.find(to);
66  if (to_it != m_storage.end())
67  throw duplicated_tag_error("A module already produced the tag '" + to.toString() + "'");
68 
69  m_storage[to] = m_storage[from];
70 }
