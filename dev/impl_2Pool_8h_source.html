58 template <typename T> std::shared_ptr<const T> Pool::raw_get(const InputTag& tag) const {
59 
60  auto it = m_storage.find(tag);
61  if (it == m_storage.end())
62  throw tag_not_found_error("No such tag in pool: " + tag.toString());
63 
64  PoolContent& v = it->second;
65 
66  try {
67  std::shared_ptr<T>& ptr = momemta::any_cast<std::shared_ptr<T>&>(v.ptr);
68 
69  return std::const_pointer_cast<const T>(ptr);
70  } catch (momemta::bad_any_cast e) {
71  LOG(fatal) << "Exception while trying to get pool content for '" << tag.toString() << "'. Requested a '"
72  << demangle(typeid(std::shared_ptr<T>).name())
73  << "' while parameter is a '"
74  << demangle(v.ptr.type().name())
75  << "'";
76  throw e;
77  }

80 template <typename T, typename... Args> std::shared_ptr<T> Pool::put(const InputTag& tag, Args&&... args) {
81  auto it = m_storage.find(tag);
82  if (it != m_storage.end()) {
83  if (it->second.valid)
84  throw duplicated_tag_error("A module already produced the tag '" + tag.toString() + "'");
85  // A module already requested this block in read-mode. This will only work if the block does not require a non-trivial constructor:
86  if (sizeof...(Args))
87  throw constructor_tag_error("A module already requested the tag '" + tag.toString()
88  + "' which seems to require a constructor call. This is currently not supported.");
89  // Since the memory is allocated, simply consider the block as valid.
90  it->second.valid = true;
91 
92  // If the block is empty, it's a delayed instantiation. Simply flag the block as valid, and allocate memory for it
93  if (it->second.ptr.empty()) {
94  auto ptr = std::make_shared<T>(std::forward<Args>(args)...);
95  it->second.ptr = momemta::any(ptr);
96  }
97 
98  } else {
99  it = create<T, Args ...>(tag, true, std::forward<Args>(args)...);
100  }
101 
102  // Update current module description
103  Description& description = get_description();
104  description.outputs.push_back(tag.parameter);
105 
106  return momemta::any_cast<std::shared_ptr<T>>(it->second.ptr);
107 }
