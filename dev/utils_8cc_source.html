44  LazyFunction::LazyFunction(lua_State* L, int index): Lazy(L) {
45  auto absolute_index = get_index(L, index);
46 
47  // Duplicate the function on the top of the stack. This ensure the stack size won't change
48  lua_pushvalue(L, absolute_index);
49 
50  // Pop the anonymous function from the stack, and store it in the global lua registry
51  ref_index = luaL_ref(L, LUA_REGISTRYINDEX);
52  }

54  momemta::any LazyFunction::operator() () const {
55 
56  LOG(trace) << "[LazyFunction::operator()] >> stack size = " << lua_gettop(L);
57 
58  // Pop the anonymous function from the registry, and push it on the top of the stack
59  lua_rawgeti(L, LUA_REGISTRYINDEX, ref_index);
60 
61  // Call the function. The function removed from the stack, and the return value is pushed on the top of the stack
62  auto result = lua_pcall(L, 0, 1, 0);
63  if (result != LUA_OK) {
64  std::string error = lua_tostring(L, -1);
65  LOG(fatal) << "Fail to call lua anonymous function. Return value is " << result << ". Error message: " << error;
66  }
67 
68  momemta::any value;
69  bool lazy = false;
70  std::tie(value, lazy) = to_any(L, -1);
71  assert(!lazy);
72 
73  lua_pop(L, 1);
74 
75  LOG(trace) << "[LazyFunction::operator()] << stack size = " << lua_gettop(L);
76 
77  return value;
78  }

80  Type type(lua_State* L, int index) {
81  int t = lua_type(L, index);
82 
83  switch (t) {
84  case LUA_TBOOLEAN:
85  return BOOLEAN;
86  break;
87 
88  case LUA_TSTRING: {
89  std::string value = lua_tostring(L, index);
90  if (InputTag::isInputTag(value))
91  return INPUT_TAG;
92  else
93  return STRING;
94  } break;
95 
96  case LUA_TNUMBER:
97  if (lua_isinteger(L, index))
98  return INTEGER;
99  else
100  return REAL;
101  break;
102 
103  case LUA_TTABLE: {
104  // We only support ParameterSet for table
105  if (lua_is_array(L, index) == -1) {
106  return PARAMETER_SET;
107  }
108 
109  } break;
110  }
111 
112  return NOT_SUPPORTED;
113  }

124  int lua_is_array(lua_State* L, int index) {
125 
126  LOG(trace) << "[lua_is_array] >> stack size = " << lua_gettop(L);
127 
128  size_t table_index = get_index(L, index);
129 
130  if (lua_type(L, table_index) != LUA_TTABLE)
131  return -1;
132 
133  size_t size = 0;
134 
135  lua_pushnil(L);
136  while (lua_next(L, table_index) != 0) {
137  if (lua_type(L, -2) != LUA_TNUMBER) {
138  lua_pop(L, 2);
139  LOG(trace) << "[lua_is_array] << stack size = " << lua_gettop(L);
140  return -1;
141  }
142 
143  size++;
144 
145  lua_pop(L, 1);
146  }
147 
148  LOG(trace) << "[lua_is_array] << stack size = " << lua_gettop(L);
149  return size;
150  }

157  Type lua_array_unique_type(lua_State* L, int index) {
158 
159  size_t absolute_index = get_index(L, index);
160 
161  if (lua_type(L, absolute_index) != LUA_TTABLE)
162  return NOT_SUPPORTED;
163 
164  Type result = NOT_SUPPORTED;
165 
166  lua_pushnil(L);
167  while (lua_next(L, absolute_index) != 0) {
168  if (result == NOT_SUPPORTED) {
169  result = type(L, -1);
170  } else {
171  Type entry_type = type(L, -1);
172 
173  if ((result == INTEGER) && (entry_type == REAL))
174  result = REAL;
175  else if ((result == REAL) && (entry_type == INTEGER))
176  result = REAL;
177  else if (result != entry_type) {
178  lua_pop(L, 2);
179  result = NOT_SUPPORTED;
180  break;
181  }
182  }
183 
184  lua_pop(L, 1);
185  }
186 
187  return result;
188  }

190  std::pair<momemta::any, bool> to_any(lua_State* L, int index) {
191 
192  LOG(trace) << "[to_any] >> stack size = " << lua_gettop(L);
193  size_t absolute_index = get_index(L, index);
194 
195  momemta::any result;
196  bool lazy = false;
197 
198  auto type = lua_type(L, absolute_index);
199  switch (type) {
200  case LUA_TNUMBER: {
201  if (lua_isinteger(L, absolute_index)) {
202  int64_t number = lua_tointeger(L, absolute_index);
203  result = number;
204  } else {
205  double number = lua_tonumber(L, absolute_index);
206  result = number;
207  }
208  } break;
209 
210  case LUA_TBOOLEAN: {
211  bool value = lua_toboolean(L, absolute_index);
212  result = value;
213  } break;
214 
215  case LUA_TSTRING: {
216  std::string value = lua_tostring(L, absolute_index);
217  if (InputTag::isInputTag(value)) {
218  InputTag tag = InputTag::fromString(value);
219  result = tag;
220  } else {
221  result = value;
222  }
223  } break;
224 
225  case LUA_TTABLE: {
226  LOG(trace) << "[to_any::table] >> stack size = " << lua_gettop(L);
227  if (lua::lua_is_array(L, absolute_index) > 0) {
228 
229  Type type = NOT_SUPPORTED;
230 
231  if ((type = lua::lua_array_unique_type(L, absolute_index)) != NOT_SUPPORTED) {
232  result = to_vector(L, absolute_index, type);
233  } else {
234  throw invalid_array_error("Various types stored into the array. This is not supported.");
235  }
236 
237  } else {
238  ParameterSet cfg;
239  ParameterSetParser::parse(cfg, L, absolute_index);
240  result = cfg;
241  }
242  LOG(trace) << "[to_any::table] << stack size = " << lua_gettop(L);
243  } break;
244 
245  case LUA_TFUNCTION: {
246  LOG(trace) << "[to_any::function] >> stack size = " << lua_gettop(L);
247 
248  result = LazyFunction(L, absolute_index);
249  lazy = true;
250 
251  LOG(trace) << "[to_any::function] << stack size = " << lua_gettop(L);
252  } break;
253 
254  case LUA_TUSERDATA: {
255  result = get_custom_type_ptr(L, absolute_index);
256 
257  } break;
258 
259  default: {
260  LOG(fatal) << "Unsupported lua type: " << lua_type(L, absolute_index);
261  throw lua::invalid_configuration_file("");
262  } break;
263  }
264 
265  LOG(trace) << "[to_any] << final type = " << demangle(result.type().name());
266  LOG(trace) << "[to_any] << stack size = " << lua_gettop(L);
267  return {result, lazy};
268  }

270  void push_any(lua_State* L, const momemta::any& value) {
271  LOG(trace) << "[push_any] >> stack size = " << lua_gettop(L);
272 
273  if (value.type() == typeid(int64_t)) {
274  int64_t v = momemta::any_cast<int64_t>(value);
275  lua_pushinteger(L, v);
276  } else if (value.type() == typeid(double)) {
277  double v = momemta::any_cast<double>(value);
278  lua_pushnumber(L, v);
279  } else if (value.type() == typeid(bool)) {
280  bool v = momemta::any_cast<bool>(value);
281  lua_pushboolean(L, v);
282  } else if (value.type() == typeid(std::string)) {
283  auto v = momemta::any_cast<std::string>(value);
284  lua_pushstring(L, v.c_str());
285  } else if (value.type() == typeid(InputTag)) {
286  auto v = momemta::any_cast<InputTag>(value).toString();
287  lua_pushstring(L, v.c_str());
288  } else {
289  LOG(fatal) << "Unsupported C++ value: " << demangle(value.type().name());
290  throw lua::unsupported_type_error(demangle(value.type().name()));
291  }
292 
293  LOG(trace) << "[push_any] << stack size = " << lua_gettop(L);
294  }

331  int module_table_newindex(lua_State* L) {
332 
333  lua_getmetatable(L, 1);
334  lua_getfield(L, -1, "__type");
335 
336  const char* module_type = luaL_checkstring(L, -1);
337  const char* module_name = luaL_checkstring(L, 2);
338 
339  // Remove field name from stack
340  lua_pop(L, 1);
341 
342  lua_getfield(L, -1, "__ptr");
343  void* cfg_ptr = lua_touserdata(L, -1);
344  ILuaCallback* callback = static_cast<ILuaCallback*>(cfg_ptr);
345 
346  callback->onModuleDeclared(module_type, module_name);
347 
348  // Remove metatable and field name from stack
349  lua_pop(L, 2);
350 
351  // And actually set the value to the table
352  lua_rawset(L, 1);
353 
354  return 0;
355  }

357  void register_modules(lua_State* L, void* ptr) {
358  std::vector<std::string> modules = ModuleFactory::get().getPluginsList();
359  for (const auto& module: modules) {
360  const char* module_name = module.c_str();
361 
362  int type = lua_getglobal(L, module_name);
363  lua_pop(L, 1);
364  if (type != LUA_TNIL) {
365  // Global already exists
366  continue;
367  }
368 
369  // Create a new empty table
370  lua_newtable(L);
371 
372  std::string module_metatable = module + "_mt";
373 
374  // Create the associated metatable
375  luaL_newmetatable(L, module_metatable.c_str());
376 
377  lua_pushstring(L, module_name);
378  lua_setfield(L, -2, "__type");
379 
380  lua_pushlightuserdata(L, ptr);
381  lua_setfield(L, -2, "__ptr");
382 
383  // Set the metadata '__newindex' function
384  const luaL_Reg l[] = {
385  {"__newindex", lua::module_table_newindex},
386  {nullptr, nullptr}
387  };
388  luaL_setfuncs(L, l, 0);
389 
390  lua_setmetatable(L, -2);
391 
392  // And register it as a global variable
393  lua_setglobal(L, module_name);
394 
395  LOG(trace) << "Registered new lua global variable '" << module_name << "'";
396  }
397  }

433  int n = lua_gettop(L);

436  }

440 
441  for(size_t i = 1; i <= size_t(n); i++) {
442  std::string input_tag = luaL_checkstring(L, i);

444  luaL_error(L, "'%s' is not a valid InputTag", input_tag.c_str());
