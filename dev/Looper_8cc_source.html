131  virtual Status work() override {
132  particles->clear();
133 
134  CALL(beginLoop);
135 
136  auto status = Status::OK;
137 
138  // For each solution, loop over all the modules
139  for (const auto& s: *solutions) {
140  if (!s.valid)
141  continue;
142 
143  *particles = s.values;
144  *jacobian = s.jacobian;
145 
146  for (auto& m: path.modules()) {
147  auto module_status = m->work();
148 
149  if (module_status == Status::OK)
150  continue;
151  else if (module_status == Status::NEXT)
152  break;
153  else {
154  status = module_status;
155  break;
156  }
157  }
158 
159  if (status != Status::OK)
160  break;
161  }
162 
163  CALL(endLoop);
164 
165  return status;
166  }
